        // 🔐 Generador de Key encriptado (simula seguridad)
        function generateSecureKey() {
            const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';
            const parts = [];
            for(let i = 0; i < 4; i++) {
                let part = '';
                for(let j = 0; j < 5; j++) {
                    part += chars[Math.floor(Math.random() * chars.length)];
                }
                parts.push(part);
            }
            // Formato: MVS-XXXXX-XXXXX-XXXXX-XXXXX
            const rawKey = 'MVS-' + parts.join('-');
            
            // "Encriptación" simple (simula ofuscación)
            let encrypted = '';
            for(let i = 0; i < rawKey.length; i++) {
                encrypted += String.fromCharCode(rawKey.charCodeAt(i) ^ 0x05);
            }
            return { raw: rawKey, encrypted: btoa(encrypted) };
        }
        
        function formatDate(timestamp) {
            const date = new Date(timestamp);
            return date.toLocaleString('es-ES', {
                year: 'numeric',
                month: '2-digit',
                day: '2-digit',
                hour: '2-digit',
                minute: '2-digit'
            });
        }
